include(boomerang-utils)

set(TESTS
    HpSomBinaryLoaderTest
    MicroX86DisTest
)

set(TESTS_WITH_ELF
    ElfBinaryLoaderTest
    SparcBinaryLoaderTest
)

set(TESTS_WITH_WIN32
    Win32BinaryLoaderTest
)


foreach(t ${TESTS})
	BOOMERANG_ADD_TEST(
		NAME ${t}
		SOURCES ${t}.h ${t}.cpp
		LIBRARIES

			boomerang
			${CMAKE_THREAD_LIBS_INIT}
	)
endforeach()

            LIBRARIES
                ${DEBUG_LIB}
                boomerang
                ${CMAKE_THREAD_LIBS_INIT}
        )
    endforeach()
endif (BOOMERANG_BUILD_LOADER_Win32)
