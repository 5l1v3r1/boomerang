
set(boomerang-sources "")

add_subdirectory(core)
add_subdirectory(db)

    LIBRARY DESTINATION lib/
    RUNTIME DESTINATION bin/
)

install(DIRECTORY . DESTINATION include/boomerang
    FILES_MATCHING PATTERN "*.h"
)
